GEDCOM 5.5.5 File Structure Grammar

GEDCOM_FILE:= [U+FEFF] + <GEDCOM_HEADER> + {<TOP_LEVEL_RECORD>} + <GEDCOM_TRAILER>
Note: U+FEFF is the optional Byte Order Mark (BOM)[cite: 13992].
GEDCOM_HEADER:= 0 HEAD +1 GEDC +2 VERS <GEDCOM_VERSION_NUMBER> +2 FORM <GEDCOM_FORM> +3 VERS <VERSION_NUMBER> +1 CHAR <CHARACTER_ENCODING> [+1 SOUR <SOURCE_SYSTEM> [+2 VERS <VERSION_NUMBER>] [+2 NAME <NAME_OF_PRODUCT>] [+2 CORP <NAME_OF_BUSINESS> [+3 <ADDRESS_STRUCTURE>] ] ] [+1 DEST <RECEIVING_SYSTEM_NAME>] [+1 DATE <TRANSMISSION_DATE> [+2 TIME <TIME_VALUE>] ] [+1 SUBM @<XREF:SUBM>@] [+1 FILE <FILE_NAME>] [+1 COPR <COPYRIGHT_GEDCOM_FILE>] [+1 LANG <LANGUAGE_ID>] [+1 NOTE <USER_TEXT> {+2 [CONC|CONT] <USER_TEXT>} ]
TOP_LEVEL_RECORD:= [ <INDI_RECORD> | <FAM_RECORD> | <OBJE_RECORD> | <NOTE_RECORD> | <REPO_RECORD> | <SOUR_RECORD> | <SUBM_RECORD> ]
INDI_RECORD:= 0 @<XREF:INDI>@ INDI {+1 <INDIVIDUAL_ATTRIBUTE>} {+1 <INDIVIDUAL_EVENT>} {+1 <FAMC_LINK>} {+1 <FAMS_LINK>} {+1 <SOURCE_CITATION>} {+1 <MULTIMEDIA_LINK>} {+1 <NOTE_STRUCTURE>} [+1 REFN <USER_REFERENCE_NUMBER>] [+1 RIN <AUTOMATED_RECORD_ID>] [+1 CHAN <CHANGE_DATE>]
FAM_RECORD:= 0 @<XREF:FAM>@ FAM {+1 <FAMILY_EVENT>} {+1 HUSB @<XREF:INDI>@} {+1 WIFE @<XREF:INDI>@} {+1 CHIL @<XREF:INDI>@} {+1 <SOURCE_CITATION>} {+1 <MULTIMEDIA_LINK>} {+1 <NOTE_STRUCTURE>} [+1 REFN <USER_REFERENCE_NUMBER>] [+1 RIN <AUTOMATED_RECORD_ID>] [+1 CHAN <CHANGE_DATE>]
SOUR_RECORD:= 0 @<XREF:SOUR>@ SOUR [+1 DATA [+2 EVEN <EVENTS_RECORDED> [+3 DATE <DATE_PERIOD>] [+3 PLAC <SOURCE_JURISDICTION_PLACE>] ] [+2 AGNC <RESPONSIBLE_AGENCY>] {<NOTE_STRUCTURE>} ] [+1 AUTH <SOURCE_ORIGINATOR> {+2 [CONC|CONT] <SOURCE_ORIGINATOR>} ] [+1 TITL <SOURCE_DESCRIPTIVE_TITLE> {+2 [CONC|CONT] <SOURCE_DESCRIPTIVE_TITLE>} ] [+1 ABBR <SOURCE_FILED_BY_ENTRY>] [+1 PUBL <SOURCE_PUBLICATION_FACTS> {+2 [CONC|CONT] <SOURCE_PUBLICATION_FACTS>} ] [+1 TEXT <TEXT_FROM_SOURCE> {+2 [CONC|CONT] <TEXT_FROM_SOURCE>} ] {<SOURCE_REPOSITORY_CITATION>} {<MULTIMEDIA_LINK>} {<NOTE_STRUCTURE>} [+1 REFN <USER_REFERENCE_NUMBER>] [+1 RIN <AUTOMATED_RECORD_ID>] [+1 CHAN <CHANGE_DATE>]
SUBM_RECORD:= 0 @<XREF:SUBM>@ SUBM +1 NAME <SUBMITTER_NAME> [+1 <ADDRESS_STRUCTURE>] {<MULTIMEDIA_LINK>} [+1 LANG <LANGUAGE_ID>] [+1 RIN <AUTOMATED_RECORD_ID>] [+1 CHAN <CHANGE_DATE>]
GEDCOM_TRAILER:= 0 TRLR
Signals the end of the GEDCOM file.
