GEDCOM 5.5.5 Formal Grammar

gedcom_file: record {record}
record: line {line}
line: level [delim [xref_id]] delim tag [delim line_value] terminator
level: [ "0" .. "9" ] { [ "0" .. "9" ] }
delim: space
tag: alphanum {alphanum}
line_value: pointer | line_item
xref_id: "@" alphanum {alphanum} "@"
pointer: "@" alphanum {alphanum} "@"
line_item: [any_char_except_terminator_or_at_sign] {any_char_except_terminator}
terminator: carriage_return | line_feed | carriage_return line_feed | line_feed carriage_return
alphanum: alpha | digit | "_"
alpha: ["A".."Z"] | ["a".."z"]
digit: ["0".."9"]
space: U+0020
any_char: Any Unicode character except control characters (U+0000 to U+001F) other than Tab (U+0009)
